25int main()
26{
27  Solver slv;
28  slv.setLogic("QF_BV"); // Set the logic
29
30  Sort bitvector32 = slv.mkBitVectorSort(32);
31
32  Term x = slv.mkConst(bitvector32, "a");
33
34  Op ext_31_1 = slv.mkOp(BITVECTOR_EXTRACT, 31, 1);
35  Term x_31_1 = slv.mkTerm(ext_31_1, x);
36
37  Op ext_30_0 = slv.mkOp(BITVECTOR_EXTRACT, 30, 0);
38  Term x_30_0 = slv.mkTerm(ext_30_0, x);
39
40  Op ext_31_31 = slv.mkOp(BITVECTOR_EXTRACT, 31, 31);
41  Term x_31_31 = slv.mkTerm(ext_31_31, x);
42
43  Op ext_0_0 = slv.mkOp(BITVECTOR_EXTRACT, 0, 0);
44  Term x_0_0 = slv.mkTerm(ext_0_0, x);
45
46  Term eq = slv.mkTerm(EQUAL, x_31_1, x_30_0);
47  cout << " Asserting: " << eq << endl;
48  slv.assertFormula(eq);
49
50  Term eq2 = slv.mkTerm(EQUAL, x_31_31, x_0_0);
51  cout << " Check entailment assuming: " << eq2 << endl;
52  cout << " Expect ENTAILED. " << endl;
53  cout << " cvc5: " << slv.checkEntailed(eq2) << endl;
54
55  return 0;
56}

22if __name__ == "__main__":
23    slv = Solver()

26    bitvector32 = slv.mkBitVectorSort(32)
27
28    x = slv.mkConst(bitvector32, "a")
29
30    ext_31_1 = slv.mkOp(BVExtract, 31, 1)
31    x_31_1 = slv.mkTerm(ext_31_1, x)
32
33    ext_30_0 = slv.mkOp(BVExtract, 30, 0)
34    x_30_0 = slv.mkTerm(ext_30_0, x)
35
36    ext_31_31 = slv.mkOp(BVExtract, 31, 31)
37    x_31_31 = slv.mkTerm(ext_31_31, x)
38
39    ext_0_0 = slv.mkOp(BVExtract, 0, 0)
40    x_0_0 = slv.mkTerm(ext_0_0, x)
41
42    # test getting indices
43    assert ext_30_0.getIndices() == (30, 0)
44
45    eq = slv.mkTerm(Equal, x_31_1, x_30_0)
46    print("Asserting:", eq)
47    slv.assertFormula(eq)
48
49    eq2 = slv.mkTerm(Equal, x_31_31, x_0_0)
50    print("Check entailment assuming:", eq2)
51    print("Expect ENTAILED")
52    print("cvc5:", slv.checkEntailed(eq2))
