 12 *
 13 * An example of solving floating-point problems with CVC4's Java API
 14 *
 15 * This example shows how to check whether CVC4 was built with floating-point
 16 * support, how to create floating-point types, variables and expressions, and
 17 * how to create rounding mode constants by solving toy problems. The example
 18 * also shows making special values (such as NaN and +oo) and converting an
 19 * IEEE 754-2008 bit-vector to a floating-point number.
 20 */
 21
 22import edu.stanford.CVC4.*;
 23import java.util.Iterator;
 24
 25public class FloatingPointArith {
 26  public static void main(String[] args) {
 27    System.loadLibrary("cvc4jni");
 28
 29    // Test whether CVC4 was built with floating-point support
 30    ExprManager em = new ExprManager();
 31    SmtEngine smt = new SmtEngine(em);
 32
 33    // Enable the model production
 34    smt.setOption("produce-models", new SExpr(true));
 35
 36    // Make single precision floating-point variables
 37    FloatingPointType fpt32 = em.mkFloatingPointType(8, 24);
 38    Expr a = em.mkVar("a", fpt32);
 39    Expr b = em.mkVar("b", fpt32);
 40    Expr c = em.mkVar("c", fpt32);
 41    Expr d = em.mkVar("d", fpt32);
 42    Expr e = em.mkVar("e", fpt32);
 43
 44    // Assert that floating-point addition is not associative:
 45    // (a + (b + c)) != ((a + b) + c)
 46    Expr rm = em.mkConst(RoundingMode.roundNearestTiesToEven);
 47    Expr lhs = em.mkExpr(Kind.FLOATINGPOINT_ADD,
 48        rm,
 49        a,
 50        em.mkExpr(Kind.FLOATINGPOINT_ADD, rm, b, c));
 51    Expr rhs = em.mkExpr(Kind.FLOATINGPOINT_ADD,
 52        rm,
 53        em.mkExpr(Kind.FLOATINGPOINT_ADD, rm, a, b),
 54        c);
 55    smt.assertFormula(em.mkExpr(Kind.NOT, em.mkExpr(Kind.EQUAL, a, b)));
 56
 57    Result r = smt.checkSat(); // result is sat
 58    assert r.isSat() == Result.Sat.SAT;
 59
 60    System.out.println("a = " + smt.getValue(a));
 61    System.out.println("b = " + smt.getValue(b));
 62    System.out.println("c = " + smt.getValue(c));
 63
 64    // Now, let's restrict `a` to be either NaN or positive infinity
 65    FloatingPointSize fps32 = new FloatingPointSize(8, 24);
 66    Expr nan = em.mkConst(FloatingPoint.makeNaN(fps32));
 67    Expr inf = em.mkConst(FloatingPoint.makeInf(fps32, /* sign */ true));
 68    smt.assertFormula(em.mkExpr(
 69        Kind.OR, em.mkExpr(Kind.EQUAL, a, inf), em.mkExpr(Kind.EQUAL, a, nan)));
 70
 71    r = smt.checkSat(); // result is sat
 72    assert r.isSat() == Result.Sat.SAT;
 73
 74    System.out.println("a = " + smt.getValue(a));
 75    System.out.println("b = " + smt.getValue(b));
 76    System.out.println("c = " + smt.getValue(c));
 77
 78    // And now for something completely different. Let's try to find a (normal)
 79    // floating-point number that rounds to different integer values for
 80    // different rounding modes.
 81    Expr rtp = em.mkConst(RoundingMode.roundTowardPositive);
 82    Expr rtn = em.mkConst(RoundingMode.roundTowardNegative);
 83    Expr op = em.mkConst(new FloatingPointToSBV(16)); // (_ fp.to_sbv 16)
 84    lhs = em.mkExpr(op, rtp, d);
 85    rhs = em.mkExpr(op, rtn, d);
 86    smt.assertFormula(em.mkExpr(Kind.FLOATINGPOINT_ISN, d));
 87    smt.assertFormula(em.mkExpr(Kind.NOT, em.mkExpr(Kind.EQUAL, lhs, rhs)));
 88
 89    r = smt.checkSat(); // result is sat
 90    assert r.isSat() == Result.Sat.SAT;
 91
 92    // Convert the result to a rational and print it
 93    Expr val = smt.getValue(d);
 94    Rational realVal =
 95        val.getConstFloatingPoint().convertToRationalTotal(new Rational(0));
 96    System.out.println("d = " + val + " = " + realVal);
 97    System.out.println("((_ fp.to_sbv 16) RTP d) = " + smt.getValue(lhs));
 98    System.out.println("((_ fp.to_sbv 16) RTN d) = " + smt.getValue(rhs));
 99
100    // For our final trick, let's try to find a floating-point number between
101    // positive zero and the smallest positive floating-point number
102    Expr zero = em.mkConst(FloatingPoint.makeZero(fps32, /* sign */ true));
103    Expr smallest =
104        em.mkConst(new FloatingPoint(8, 24, new BitVector(32, 0b001)));
105    smt.assertFormula(em.mkExpr(Kind.AND,
106        em.mkExpr(Kind.FLOATINGPOINT_LT, zero, e),
107        em.mkExpr(Kind.FLOATINGPOINT_LT, e, smallest)));
108
109    r = smt.checkSat(); // result is unsat
110    assert r.isSat() == Result.Sat.UNSAT;
111  }
112}

14# A simple demonstration of the solving capabilities of the cvc5
15# floating point solver through the Python API contributed by Eva
16# Darulova. This requires building cvc5 with symfpu.
17##
18
19import pycvc5
20from pycvc5 import kinds
21
22if __name__ == "__main__":
23    slv = pycvc5.Solver()
24
25    slv.setOption("produce-models", "true")
26    slv.setLogic("QF_FP")
27
28    # single 32-bit precision
29    fp32 = slv.mkFloatingPointSort(8, 24)
30
31    # the standard rounding mode
32    rm = slv.mkRoundingMode(pycvc5.RoundNearestTiesToEven)
33
34    # create a few single-precision variables
35    x = slv.mkConst(fp32, 'x')
36    y = slv.mkConst(fp32, 'y')
37    z = slv.mkConst(fp32, 'z')
38
39    # check floating-point arithmetic is commutative, i.e. x + y == y + x
40    commutative = slv.mkTerm(kinds.FPEq, slv.mkTerm(kinds.FPAdd, rm, x, y), slv.mkTerm(kinds.FPAdd, rm, y, x))
41
42    slv.push()
43    slv.assertFormula(slv.mkTerm(kinds.Not, commutative))
44    print("Checking floating-point commutativity")
45    print("Expect SAT (property does not hold for NaN and Infinities).")
46    print("cvc5:", slv.checkSat())
47    print("Model for x:", slv.getValue(x))
48    print("Model for y:", slv.getValue(y))
49
50    # disallow NaNs and Infinities
51    slv.assertFormula(slv.mkTerm(kinds.Not, slv.mkTerm(kinds.FPIsNan, x)))
52    slv.assertFormula(slv.mkTerm(kinds.Not, slv.mkTerm(kinds.FPIsInf, x)))
53    slv.assertFormula(slv.mkTerm(kinds.Not, slv.mkTerm(kinds.FPIsNan, y)))
54    slv.assertFormula(slv.mkTerm(kinds.Not, slv.mkTerm(kinds.FPIsInf, y)))
55
56    print("Checking floating-point commutativity assuming x and y are not NaN or Infinity")
57    print("Expect UNSAT.")
58    print("cvc5:", slv.checkSat())
59
60    # check floating-point arithmetic is not associative
61    slv.pop()
62
63    # constrain x, y, z between -3.14 and 3.14 (also disallows NaN and infinity)
64    a = slv.mkFloatingPoint(8, 24, slv.mkBitVector("11000000010010001111010111000011", 2))  # -3.14
65    b = slv.mkFloatingPoint(8, 24, slv.mkBitVector("01000000010010001111010111000011", 2))  # 3.14
66
67    bounds_x = slv.mkTerm(kinds.And, slv.mkTerm(kinds.FPLeq, a, x), slv.mkTerm(kinds.FPLeq, x, b))
68    bounds_y = slv.mkTerm(kinds.And, slv.mkTerm(kinds.FPLeq, a, y), slv.mkTerm(kinds.FPLeq, y, b))
69    bounds_z = slv.mkTerm(kinds.And, slv.mkTerm(kinds.FPLeq, a, z), slv.mkTerm(kinds.FPLeq, z, b))
70    slv.assertFormula(slv.mkTerm(kinds.And, slv.mkTerm(kinds.And, bounds_x, bounds_y), bounds_z))
71
72    # (x + y) + z == x + (y + z)
73    lhs = slv.mkTerm(kinds.FPAdd, rm, slv.mkTerm(kinds.FPAdd, rm, x, y), z)
74    rhs = slv.mkTerm(kinds.FPAdd, rm, x, slv.mkTerm(kinds.FPAdd, rm, y, z))
75    associative = slv.mkTerm(kinds.Not, slv.mkTerm(kinds.FPEq, lhs, rhs))
76
77    slv.assertFormula(associative)
78
79    print("Checking floating-point associativity")
80    print("Expect SAT.")
81    print("cvc5:", slv.checkSat())
82    print("Model for x:", slv.getValue(x))
83    print("Model for y:", slv.getValue(y))
84    print("Model for z:", slv.getValue(z))
