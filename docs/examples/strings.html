12 *
13 * A simple demonstration of reasoning about strings with CVC4 via C++ API.
14 */
15
16#include <cvc5/cvc5.h>
17
18#include <iostream>
19
20using namespace cvc5::api;
21
22int main()
23{
24  Solver slv;
25
26  // Set the logic
27  slv.setLogic("QF_SLIA");
28  // Produce models
29  slv.setOption("produce-models", "true");
30  // The option strings-exp is needed
31  slv.setOption("strings-exp", "true");
32  // Set output language to SMTLIB2
33  slv.setOption("output-language", "smt2");
34
35  // String type
36  Sort string = slv.getStringSort();
37
38  // std::string
39  std::string str_ab("ab");
40  // String constants
41  Term ab  = slv.mkString(str_ab);
42  Term abc = slv.mkString("abc");
43  // String variables
44  Term x = slv.mkConst(string, "x");
45  Term y = slv.mkConst(string, "y");
46  Term z = slv.mkConst(string, "z");
47
48  // String concatenation: x.ab.y
49  Term lhs = slv.mkTerm(STRING_CONCAT, x, ab, y);
50  // String concatenation: abc.z
51  Term rhs = slv.mkTerm(STRING_CONCAT, abc, z);
52  // x.ab.y = abc.z
53  Term formula1 = slv.mkTerm(EQUAL, lhs, rhs);
54
55  // Length of y: |y|
56  Term leny = slv.mkTerm(STRING_LENGTH, y);
57  // |y| >= 0
58  Term formula2 = slv.mkTerm(GEQ, leny, slv.mkInteger(0));
59
60  // Regular expression: (ab[c-e]*f)|g|h
61  Term r = slv.mkTerm(REGEXP_UNION,
62    slv.mkTerm(REGEXP_CONCAT,
63      slv.mkTerm(STRING_TO_REGEXP, slv.mkString("ab")),
64      slv.mkTerm(REGEXP_STAR,
65        slv.mkTerm(REGEXP_RANGE, slv.mkString("c"), slv.mkString("e"))),
66      slv.mkTerm(STRING_TO_REGEXP, slv.mkString("f"))),
67    slv.mkTerm(STRING_TO_REGEXP, slv.mkString("g")),
68    slv.mkTerm(STRING_TO_REGEXP, slv.mkString("h")));
69
70  // String variables
71  Term s1 = slv.mkConst(string, "s1");
72  Term s2 = slv.mkConst(string, "s2");
73  // String concatenation: s1.s2
74  Term s = slv.mkTerm(STRING_CONCAT, s1, s2);
75
76  // s1.s2 in (ab[c-e]*f)|g|h
77  Term formula3 = slv.mkTerm(STRING_IN_REGEXP, s, r);
78
79  // Make a query
80  Term q = slv.mkTerm(AND,
81    formula1,
82    formula2,
83    formula3);
84
85  // check sat
86  Result result = slv.checkSatAssuming(q);
87  std::cout << "CVC4 reports: " << q << " is " << result << "." << std::endl;
88
89  if(result.isSat())
90  {
91    std::cout << "  x  = " << slv.getValue(x) << std::endl;
92    std::cout << "  s1.s2 = " << slv.getValue(s) << std::endl;
93  }
94}

16import edu.stanford.CVC4.*;
17
18public class Strings {
19  public static void main(String[] args) {
20    System.loadLibrary("cvc4jni");
21
22    ExprManager em = new ExprManager();
23    SmtEngine smt = new SmtEngine(em);
24
25    // Set the logic
26    smt.setLogic("S");
27
28    // Produce models
29    smt.setOption("produce-models", new SExpr(true));
30    // The option strings-exp is needed
31    smt.setOption("strings-exp", new SExpr(true));
32	// output-language
33    smt.setOption("output-language", new SExpr("smt2"));
34
35    // String type
36    Type string = em.stringType();
37
38    // String constants
39    Expr ab  = em.mkConst(new CVC4String("ab"));
40    Expr abc = em.mkConst(new CVC4String("abc"));
41    // Variables
42    Expr x = em.mkVar("x", string);
43    Expr y = em.mkVar("y", string);
44    Expr z = em.mkVar("z", string);
45
46    // String concatenation: x.ab.y
47    Expr lhs = em.mkExpr(Kind.STRING_CONCAT, x, ab, y);
48    // String concatenation: abc.z
49    Expr rhs = em.mkExpr(Kind.STRING_CONCAT, abc, z);;
50    // x.ab.y = abc.z
51    Expr formula1 = em.mkExpr(Kind.EQUAL, lhs, rhs);
52
53    // Length of y: |y|
54    Expr leny = em.mkExpr(Kind.STRING_LENGTH, y);
55    // |y| >= 0
56    Expr formula2 = em.mkExpr(Kind.GEQ, leny, em.mkConst(new Rational(0)));
57
58    // Regular expression: (ab[c-e]*f)|g|h
59    Expr r = em.mkExpr(Kind.REGEXP_UNION,
60      em.mkExpr(Kind.REGEXP_CONCAT,
61        em.mkExpr(Kind.STRING_TO_REGEXP, em.mkConst(new CVC4String("ab"))),
62        em.mkExpr(Kind.REGEXP_STAR,
63          em.mkExpr(Kind.REGEXP_RANGE, em.mkConst(new CVC4String("c")), em.mkConst(new CVC4String("e")))),
64        em.mkExpr(Kind.STRING_TO_REGEXP, em.mkConst(new CVC4String("f")))),
65      em.mkExpr(Kind.STRING_TO_REGEXP, em.mkConst(new CVC4String("g"))),
66      em.mkExpr(Kind.STRING_TO_REGEXP, em.mkConst(new CVC4String("h"))));
67
68    // String variables
69    Expr s1 = em.mkVar("s1", string);
70    Expr s2 = em.mkVar("s2", string);
71    // String concatenation: s1.s2
72    Expr s = em.mkExpr(Kind.STRING_CONCAT, s1, s2);
73
74    // s1.s2 in (ab[c-e]*f)|g|h
75    Expr formula3 = em.mkExpr(Kind.STRING_IN_REGEXP, s, r);
76
77	// Make a query
78    Expr q = em.mkExpr(Kind.AND,
79      formula1,
80      formula2,
81      formula3);
82
83     // check sat
84     Result result = smt.checkSat(q);
85     System.out.println("CVC4 reports: " + q + " is " + result + ".");
86
87     System.out.println("  x  = " + smt.getValue(x));
88     System.out.println("  s1.s2 = " + smt.getValue(s));
89  }
90}
