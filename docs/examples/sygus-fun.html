 12 *
 13 * A simple demonstration of the Sygus API.
 14 *
 15 * A simple demonstration of how to use the Sygus API to synthesize max and min
 16 * functions. Here is the same problem written in Sygus V2 format:
 17 *
 18 * (set-logic LIA)
 19 *
 20 * (synth-fun max ((x Int) (y Int)) Int
 21 *   ((Start Int) (StartBool Bool))
 22 *   ((Start Int (0 1 x y
 23 *                (+ Start Start)
 24 *                (- Start Start)
 25 *                (ite StartBool Start Start)))
 26 *    (StartBool Bool ((and StartBool StartBool)
 27 *                     (not StartBool)
 28 *                     (<= Start Start)))))
 29 *
 30 * (synth-fun min ((x Int) (y Int)) Int)
 31 *
 32 * (declare-var x Int)
 33 * (declare-var y Int)
 34 *
 35 * (constraint (>= (max x y) x))
 36 * (constraint (>= (max x y) y))
 37 * (constraint (or (= x (max x y))
 38 *                 (= y (max x y))))
 39 * (constraint (= (+ (max x y) (min x y))
 40 *                (+ x y)))
 41 *
 42 * (check-synth)
 43 *
 44 * The printed output for this example should be equivalent to:
 45 * (
 46 *   (define-fun max ((x Int) (y Int)) Int (ite (<= x y) y x))
 47 *   (define-fun min ((x Int) (y Int)) Int (ite (<= x y) x y))
 48 * )
 49 */
 50
 51#include <cvc5/cvc5.h>
 52
 53#include <iostream>
 54
 55#include "utils.h"
 56
 57using namespace cvc5::api;
 58
 59int main()
 60{
 61  Solver slv;
 62
 63  // required options
 64  slv.setOption("lang", "sygus2");
 65  slv.setOption("incremental", "false");
 66
 67  // set the logic
 68  slv.setLogic("LIA");
 69
 70  Sort integer = slv.getIntegerSort();
 71  Sort boolean = slv.getBooleanSort();
 72
 73  // declare input variables for the functions-to-synthesize
 74  Term x = slv.mkVar(integer, "x");
 75  Term y = slv.mkVar(integer, "y");
 76
 77  // declare the grammar non-terminals
 78  Term start = slv.mkVar(integer, "Start");
 79  Term start_bool = slv.mkVar(boolean, "StartBool");
 80
 81  // define the rules
 82  Term zero = slv.mkInteger(0);
 83  Term one = slv.mkInteger(1);
 84
 85  Term plus = slv.mkTerm(PLUS, start, start);
 86  Term minus = slv.mkTerm(MINUS, start, start);
 87  Term ite = slv.mkTerm(ITE, start_bool, start, start);
 88
 89  Term And = slv.mkTerm(AND, start_bool, start_bool);
 90  Term Not = slv.mkTerm(NOT, start_bool);
 91  Term leq = slv.mkTerm(LEQ, start, start);
 92
 93  // create the grammar object
 94  Grammar g = slv.mkSygusGrammar({x, y}, {start, start_bool});
 95
 96  // bind each non-terminal to its rules
 97  g.addRules(start, {zero, one, x, y, plus, minus, ite});
 98  g.addRules(start_bool, {And, Not, leq});
 99
100  // declare the functions-to-synthesize. Optionally, provide the grammar
101  // constraints
102  Term max = slv.synthFun("max", {x, y}, integer, g);
103  Term min = slv.synthFun("min", {x, y}, integer);
104
105  // declare universal variables.
106  Term varX = slv.mkSygusVar(integer, "x");
107  Term varY = slv.mkSygusVar(integer, "y");
108
109  Term max_x_y = slv.mkTerm(APPLY_UF, max, varX, varY);
110  Term min_x_y = slv.mkTerm(APPLY_UF, min, varX, varY);
111
112  // add semantic constraints
113  // (constraint (>= (max x y) x))
114  slv.addSygusConstraint(slv.mkTerm(GEQ, max_x_y, varX));
115
116  // (constraint (>= (max x y) y))
117  slv.addSygusConstraint(slv.mkTerm(GEQ, max_x_y, varY));
118
119  // (constraint (or (= x (max x y))
120  //                 (= y (max x y))))
121  slv.addSygusConstraint(slv.mkTerm(
122      OR, slv.mkTerm(EQUAL, max_x_y, varX), slv.mkTerm(EQUAL, max_x_y, varY)));
123
124  // (constraint (= (+ (max x y) (min x y))
125  //                (+ x y)))
126  slv.addSygusConstraint(slv.mkTerm(
127      EQUAL, slv.mkTerm(PLUS, max_x_y, min_x_y), slv.mkTerm(PLUS, varX, varY)));
128
129  // print solutions if available
130  if (slv.checkSynth().isUnsat())
131  {
132    // Output should be equivalent to:
133    // (
134    //   (define-fun max ((x Int) (y Int)) Int (ite (<= x y) y x))
135    //   (define-fun min ((x Int) (y Int)) Int (ite (<= x y) x y))
136    // )
137    std::vector<Term> terms = {max, min};
138    printSynthSolutions(terms, slv.getSynthSolutions(terms));
139  }
140
141  return 0;
142}
