 16import static io.github.cvc5.api.Kind.*;
 17
 18import io.github.cvc5.api.*;
 19
 20/*
 21This file uses the API to make a sat call equivalent to the following benchmark:
 22(set-logic ALL)
 23(set-option :finite-model-find true)
 24(set-option :produce-models true)
 25(set-option :sets-ext true)
 26(set-option :output-language "smt2")
 27(declare-sort Person 0)
 28(declare-fun people () (Set (Tuple Person)))
 29(declare-fun males () (Set (Tuple Person)))
 30(declare-fun females() (Set (Tuple Person)))
 31(declare-fun father () (Set (Tuple Person Person)))
 32(declare-fun mother () (Set (Tuple Person Person)))
 33(declare-fun parent () (Set (Tuple Person Person)))
 34(declare-fun ancestor () (Set (Tuple Person Person)))
 35(declare-fun descendant () (Set (Tuple Person Person)))
 36
 37(assert (= people (as univset (Set (Tuple Person)))))
 38(assert (not (= males (as emptyset (Set (Tuple Person))))))
 39(assert (not (= females (as emptyset (Set (Tuple Person))))))
 40(assert (= (intersection males females) (as emptyset (Set (Tuple Person)))))
 41
 42; father relation is not empty
 43(assert (not (= father (as emptyset (Set (Tuple Person Person))))))
 44; mother relation is not empty
 45(assert (not (= mother (as emptyset (Set (Tuple Person Person))))))
 46
 47; fathers are males
 48(assert (subset (join father people) males))
 49; mothers are females
 50(assert (subset (join mother people) females))
 51
 52; parent
 53(assert (= parent (union father mother)))
 54
 55; no self ancestor
 56(assert (forall ((x Person)) (not (member (mkTuple x x) ancestor))))
 57
 58; descendant
 59(assert (= descendant (tclosure parent)))
 60
 61; ancestor
 62(assert (= ancestor (transpose descendant)))
 63
 64(check-sat)
 65(get-model)
 66 */
 67
 68public class Relations
 69{
 70  public static void main(String[] args) throws CVC5ApiException
 71  {
 72    try (Solver solver = new Solver())
 73    {
 74      // Set the logic
 75      solver.setLogic("ALL");
 76
 77      // options
 78      solver.setOption("produce-models", "true");
 79      solver.setOption("finite-model-find", "true");
 80      solver.setOption("sets-ext", "true");
 81      solver.setOption("output-language", "smt2");
 82
 83      // (declare-sort Person 0)
 84      Sort personSort = solver.mkUninterpretedSort("Person");
 85
 86      // (Tuple Person)
 87      Sort tupleArity1 = solver.mkTupleSort(new Sort[] {personSort});
 88      // (Set (Tuple Person))
 89      Sort relationArity1 = solver.mkSetSort(tupleArity1);
 90
 91      // (Tuple Person Person)
 92      Sort tupleArity2 = solver.mkTupleSort(new Sort[] {personSort, personSort});
 93      // (Set (Tuple Person Person))
 94      Sort relationArity2 = solver.mkSetSort(tupleArity2);
 95
 96      // empty set
 97      Term emptySetTerm = solver.mkEmptySet(relationArity1);
 98
 99      // empty relation
100      Term emptyRelationTerm = solver.mkEmptySet(relationArity2);
101
102      // universe set
103      Term universeSet = solver.mkUniverseSet(relationArity1);
104
105      // variables
106      Term people = solver.mkConst(relationArity1, "people");
107      Term males = solver.mkConst(relationArity1, "males");
108      Term females = solver.mkConst(relationArity1, "females");
109      Term father = solver.mkConst(relationArity2, "father");
110      Term mother = solver.mkConst(relationArity2, "mother");
111      Term parent = solver.mkConst(relationArity2, "parent");
112      Term ancestor = solver.mkConst(relationArity2, "ancestor");
113      Term descendant = solver.mkConst(relationArity2, "descendant");
114
115      Term isEmpty1 = solver.mkTerm(EQUAL, males, emptySetTerm);
116      Term isEmpty2 = solver.mkTerm(EQUAL, females, emptySetTerm);
117
118      // (assert (= people (as univset (Set (Tuple Person)))))
119      Term peopleAreTheUniverse = solver.mkTerm(EQUAL, people, universeSet);
120      // (assert (not (= males (as emptyset (Set (Tuple Person))))))
121      Term maleSetIsNotEmpty = solver.mkTerm(NOT, isEmpty1);
122      // (assert (not (= females (as emptyset (Set (Tuple Person))))))
123      Term femaleSetIsNotEmpty = solver.mkTerm(NOT, isEmpty2);
124
125      // (assert (= (intersection males females) (as emptyset (Set (Tuple
126      // Person)))))
127      Term malesFemalesIntersection = solver.mkTerm(INTERSECTION, males, females);
128      Term malesAndFemalesAreDisjoint =
129          solver.mkTerm(EQUAL, malesFemalesIntersection, emptySetTerm);
130
131      // (assert (not (= father (as emptyset (Set (Tuple Person Person))))))
132      // (assert (not (= mother (as emptyset (Set (Tuple Person Person))))))
133      Term isEmpty3 = solver.mkTerm(EQUAL, father, emptyRelationTerm);
134      Term isEmpty4 = solver.mkTerm(EQUAL, mother, emptyRelationTerm);
135      Term fatherIsNotEmpty = solver.mkTerm(NOT, isEmpty3);
136      Term motherIsNotEmpty = solver.mkTerm(NOT, isEmpty4);
137
138      // fathers are males
139      // (assert (subset (join father people) males))
140      Term fathers = solver.mkTerm(JOIN, father, people);
141      Term fathersAreMales = solver.mkTerm(SUBSET, fathers, males);
142
143      // mothers are females
144      // (assert (subset (join mother people) females))
145      Term mothers = solver.mkTerm(JOIN, mother, people);
146      Term mothersAreFemales = solver.mkTerm(SUBSET, mothers, females);
147
148      // (assert (= parent (union father mother)))
149      Term unionFatherMother = solver.mkTerm(UNION, father, mother);
150      Term parentIsFatherOrMother = solver.mkTerm(EQUAL, parent, unionFatherMother);
151
152      // (assert (= parent (union father mother)))
153      Term transitiveClosure = solver.mkTerm(TCLOSURE, parent);
154      Term descendantFormula = solver.mkTerm(EQUAL, descendant, transitiveClosure);
155
156      // (assert (= parent (union father mother)))
157      Term transpose = solver.mkTerm(TRANSPOSE, descendant);
158      Term ancestorFormula = solver.mkTerm(EQUAL, ancestor, transpose);
159
160      // (assert (forall ((x Person)) (not (member (mkTuple x x) ancestor))))
161      Term x = solver.mkVar(personSort, "x");
162      DatatypeConstructor constructor = tupleArity2.getDatatype().getConstructor(0);
163      Term xxTuple = solver.mkTerm(APPLY_CONSTRUCTOR, constructor.getConstructorTerm(), x, x);
164      Term member = solver.mkTerm(MEMBER, xxTuple, ancestor);
165      Term notMember = solver.mkTerm(NOT, member);
166
167      Term quantifiedVariables = solver.mkTerm(BOUND_VAR_LIST, x);
168      Term noSelfAncestor = solver.mkTerm(FORALL, quantifiedVariables, notMember);
169
170      // formulas
171      solver.assertFormula(peopleAreTheUniverse);
172      solver.assertFormula(maleSetIsNotEmpty);
173      solver.assertFormula(femaleSetIsNotEmpty);
174      solver.assertFormula(malesAndFemalesAreDisjoint);
175      solver.assertFormula(fatherIsNotEmpty);
176      solver.assertFormula(motherIsNotEmpty);
177      solver.assertFormula(fathersAreMales);
178      solver.assertFormula(mothersAreFemales);
179      solver.assertFormula(parentIsFatherOrMother);
180      solver.assertFormula(descendantFormula);
181      solver.assertFormula(ancestorFormula);
182      solver.assertFormula(noSelfAncestor);
183
184      // check sat
185      Result result = solver.checkSat();
186
187      // output
188      System.out.println("Result     = " + result);
189      System.out.println("people     = " + solver.getValue(people));
190      System.out.println("males      = " + solver.getValue(males));
191      System.out.println("females    = " + solver.getValue(females));
192      System.out.println("father     = " + solver.getValue(father));
193      System.out.println("mother     = " + solver.getValue(mother));
194      System.out.println("parent     = " + solver.getValue(parent));
195      System.out.println("descendant = " + solver.getValue(descendant));
196      System.out.println("ancestor   = " + solver.getValue(ancestor));
197    }
198  }
199}
